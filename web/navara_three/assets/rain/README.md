# Raindrop textures

These textures are generated by the script below.

The texture is made by [this research](https://www.cs.columbia.edu/CAVE/publications/pdfs/Garg_TOG06.pdf).
You can get the texture from [DB](https://cave.cs.columbia.edu/repository/Rain).

```js
import fs from 'fs/promises';
import path from 'path';
import sharp from 'sharp';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

const ROOT_DIR = path.join(__dirname, 'point_light_database');
const CAMERA_ANGLES = ['00', '20', '40', '60', '80'];
const SIZES = ['size16'];
const OUTPUT_DIR = path.join(__dirname, 'output');

// PATTERN: cv{CAM}_v{VERT}_h{HORZ}_osc{N}.png
const FILENAME_REGEX = /^cv(-?\d+)_v(-?\d+)_h(-?\d+)_osc(\d+)\.png$/;

async function collectImages(sizePath) {
  const oscMap = new Map();

  for (const cam of CAMERA_ANGLES) {
    const camDir = path.join(sizePath, `dcam${cam}`);
    try {
      const files = await fs.readdir(camDir);
      for (const file of files) {
        const match = file.match(FILENAME_REGEX);
        if (!match) {
          throw new Error("Unexpected" + file);
        };

        const [, camAngle, v, h, osc] = match;
        const key = `osc${osc}`;
        if (!oscMap.has(key)) oscMap.set(key, new Map());

        const lightKey = { v: parseInt(v), h: parseInt(h), key: `v${v}_h${h}` };
        const fileMap = oscMap.get(key);

        if (!fileMap.has(cam)) fileMap.set(cam, new Map());
        fileMap.get(cam).set(lightKey.key, {
          filepath: path.join(camDir, file),
          v: lightKey.v,
          h: lightKey.h
        });
      }
    } catch(e) {
      console.warn(`⚠️ Skipped missing: ${camDir}: `, e);
    }
  }

  return oscMap;
}

async function processSize(size) {
  const sizePath = path.join(ROOT_DIR, size);
  const oscMap = await collectImages(sizePath);

  for (const [osc, camMap] of oscMap.entries()) {
    const sortedCams = CAMERA_ANGLES.filter(c => camMap.has(c));
    const lightSet = new Set();

    for (const cam of sortedCams) {
      for (const [key, { v, h }] of camMap.get(cam)) {
        lightSet.add(`${v}_${h}`);
      }
    }

    const sortedLightKeys = Array.from(lightSet)
      .map(str => {
        const [v, h] = str.split('_').map(Number);
        return { key: `v${v}_h${h}`, v, h };
      })
      .sort((a, b) => a.v - b.v || a.h - b.h);

    const rowBuffers = [];
    const rowHeights = [];

    for (const cam of sortedCams) {
      const cellImages = [];
      for (const light of sortedLightKeys) {
        const cell = camMap.get(cam).get(light.key);
        console.log(light.key);
        cellImages.push(sharp(cell.filepath).ensureAlpha());
      }

      const imageBuffers = await Promise.all(cellImages.map(img => img.toBuffer()));
      const imageMetas = await Promise.all(imageBuffers.map(buf => sharp(buf).metadata()));
      const rowHeight = Math.max(...imageMetas.map(m => m.height));
      console.log(cam, rowHeight)
      rowHeights.push(rowHeight);

      const totalWidth = imageMetas.reduce((acc, m) => acc + m.width, 0);

      let xOffset = 0;
      const composites = imageBuffers.map((buf, idx) => {
        const left = xOffset;
        xOffset += imageMetas[idx].width;
        return { input: buf, top: 0, left };
      });

      const row = await sharp({
        create: {
          width: totalWidth,
          height: rowHeight,
          channels: 4,
          background: { r: 0, g: 0, b: 0, alpha: 0 }
        }
      })
        .composite(composites)
        .png()
        .toBuffer();

      rowBuffers.push({ buffer: row, width: totalWidth, height: rowHeight });
    }

    const finalWidth = Math.max(...rowBuffers.map(r => r.width));
    const finalHeight = rowBuffers.reduce((acc, r) => acc + r.height, 0);

    let yOffset = 0;
    const finalComposites = rowBuffers.map(r => {
      const top = yOffset;
      yOffset += r.height;
      return { input: r.buffer, top, left: 0 };
    });

    await sharp({
      create: {
        width: finalWidth,
        height: finalHeight,
        channels: 4,
        background: { r: 0, g: 0, b: 0, alpha: 0 }
      }
    })
      .composite(finalComposites)
      .webp({ quality: 90 })
      .toFile(path.join(OUTPUT_DIR, `raindrop_${size}_${osc}.webp`));

    console.log(`✅ Created: ${size}_${osc}.png`);
  }
}

async function main() {
  await fs.mkdir(OUTPUT_DIR, { recursive: true });

  for (const size of SIZES) {
    console.log(`📦 Processing ${size}...`);
    await processSize(size);
  }
}

main().catch(console.error);
```
